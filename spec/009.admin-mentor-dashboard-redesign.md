## 6. Logic / Pseudocode

### 6.1 Admin Live Stats

```typescript
// DashboardAdmin.tsx
const [stats, setStats] = useState({ users: 0, mentors: 0, coins: 0 });

useEffect(() => {
  const fetchStats = async () => {
    const [usersRes, mentorsRes, coinsRes] = await Promise.all([
      fetch("http://localhost:5000/api/auth/user"),
      API.get("/mentor"),
      fetch("http://localhost:5000/api/trade/coins"),
    ]);
    const usersData = await usersRes.json();
    const coinsData = await coinsRes.json();
    setStats({
      users: usersData.data?.length || 0,
      mentors: mentorsRes.data?.length || 0,
      coins: coinsData.data?.length || 0,
    });
  };
  fetchStats();
}, []);
```

### 6.2 Mentor Live Profile

```typescript
// MentorDashboard.tsx
const [mentorName, setMentorName] = useState("");
const [studentCount, setStudentCount] = useState(0);
const [chatCount, setChatCount] = useState(0);

useEffect(() => {
  // From localStorage
  const user = JSON.parse(localStorage.getItem("user") || "{}");
  setMentorName(user.name || "Mentor");

  // Live stats
  const token = localStorage.getItem("token");
  Promise.all([
    axios.get("/api/trade/mentor/students", { headers: { Authorization: `Bearer ${token}` }}),
    axios.get("/api/chat/conversation", { headers: { Authorization: `Bearer ${token}` }}),
  ]).then(([studentsRes, chatsRes]) => {
    setStudentCount(studentsRes.data?.data?.length || 0);
    setChatCount(chatsRes.data?.data?.length || 0);
  });
}, []);
```
